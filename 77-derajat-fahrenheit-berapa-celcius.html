16. konversikan:a.77 derajat fahrenheit=......derajat celcius=......derajat reamur=......kelvin

Celcius = 5x5 = 25
Reamur = 5x4 = 20

77°F = °C

77°F = 5/9 × (t-32)°C

77°F = 5/9 × (77-32)°C

77°F = 5/9 × 45°C

77°F = 5 × 5°C

77°F = 25°C


77°F = °R

77°F = 4/9 × (t-32)°R

77°F = 4/9 × (77-32)°R

77°F = 4/9 × 45°R

77°F = 4 × 5°R

77°F = 20°R


77°F = K

77°F = 5/9 × (t-32) + 273K

77°F = 5/9 × (77-32) + 273K

77°F = 5/9 × 45 + 273K

77°F = 5 × 5 + 273K

77°F = 25 + 273K

77°F = 298K
