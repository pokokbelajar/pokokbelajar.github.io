PQ² = 676 - 100
PQ² = 576
PQ = √576
PQ = 24 cm

 PQ = √576 cm

[tex] \frac{y2 - y1}{x2 - x1} [/tex]
a.
kemiringan garis PQ =
[tex] \frac{ - 9 - 6}{7 - ( - 3)} = \frac{ - 15}{7 + 3} = \frac{ - 15}{10} = - \frac{15}{10} = - \frac{3}{2} [/tex]
kemiringan garis RS =
[tex] \frac{4 - 2}{3 - ( - 8)} = \frac{2}{3 + 8} = \frac{2}{11} [/tex]
